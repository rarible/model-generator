package com.rarible.test

import java.time.Instant
import kotlin.Int
import kotlin.Long
import kotlin.String
import kotlin.collections.List
import kotlin.collections.Map

data class SinglePrimitivesItemDto (
    val id : Long,
    val name : String,
    val index : Int,
    val count : Int,
    val date : Instant?,
    val strings : List<String>?,
    val longs : List<Long>?,
    val map : Map<String, Instant>?,
    val type : Type,
    val dataType : DataType?,
    val features : List<Features>?,
    val modes : List<Modes>
) {

    enum class Type {
        ERC721,
        ERC1155
    }

    enum class DataType {
        LEGACY,
        MODERN
    }

    enum class Features {
        APPROVE,
        BURN,
        MINT
    }

    enum class Modes {
        ONE,
        TWO,
        THREE
    }

}
