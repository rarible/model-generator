package com.rarible.test

import java.lang.Integer
import java.lang.Long
import java.time.Instant
import kotlin.String
import kotlin.collections.List
import kotlin.collections.Map

data class SinglePrimitivesItem (
    val id : Long,
    val name : String,
    val index : Integer,
    val count : Integer,
    val date : Instant?,
    val strings : List<String>?,
    val longs : List<Long>?,
    val map : Map<String, Instant>?,
    val type : Type,
    val dataType : DataType?,
    val features : List<Features>?,
    val modes : List<Modes>,
) {

    enum class Type {
        ERC721,
        ERC1155,
    }

    enum class DataType {
        LEGACY,
        MODERN,
    }

    enum class Features {
        APPROVE,
        BURN,
        MINT,
    }

    enum class Modes {
        ONE,
        TWO,
        THREE,
    }

}
