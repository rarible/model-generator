package com.rarible.test

import com.fasterxml.jackson.annotation.JsonSubTypes
import com.fasterxml.jackson.annotation.JsonTypeInfo
import java.lang.Integer
import kotlin.String

@JsonTypeInfo(use = JsonTypeInfo.Id.NAME, include = JsonTypeInfo.As.PROPERTY)
@JsonSubTypes(
    JsonSubTypes.Type(name = "SUB_ITEM_1", value = OneOfFieldSubItem1::class),
    JsonSubTypes.Type(name = "SUB_ITEM_2", value = OneOfFieldSubItem2::class),
)
sealed class OneOfFieldItem {
}

//--------------- OneOfFieldSubItem1 ---------------//
class OneOfFieldSubItem1 (
    val id : Integer,
    val name : String,
) : OneOfFieldItem()

//--------------- OneOfFieldSubItem2 ---------------//
class OneOfFieldSubItem2 (
    val id : Integer,
) : OneOfFieldItem()
