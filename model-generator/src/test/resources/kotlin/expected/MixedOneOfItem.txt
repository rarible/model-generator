package com.rarible.test

import com.fasterxml.jackson.annotation.JsonSubTypes
import com.fasterxml.jackson.annotation.JsonTypeInfo
import com.rarible.test.MixedPartDto
import kotlin.String
import kotlin.collections.List

@JsonTypeInfo(use = JsonTypeInfo.Id.NAME, include = JsonTypeInfo.As.PROPERTY)
@JsonSubTypes(
    JsonSubTypes.Type(name = "SUB_ITEM_1", value = MixedSubItem1Dto::class),
    JsonSubTypes.Type(name = "SUB_ITEM_2", value = MixedSubItem2Dto::class),
)
sealed class MixedOneOfItemDto {
    abstract val parts : List<MixedPartDto>
}

//--------------- MixedSubItem1Dto ---------------//
class MixedSubItem1Dto (
    override val parts : List<MixedPartDto>,
    val modes : List<Modes>,
    val features : List<Features>?,
) : MixedOneOfItemDto() {

    enum class Modes {
        ONE,
        TWO,
        THREE,
    }

    enum class Features {
        APPROVE_FOR_ALL,
        SET_URI_PREFIX,
        BURN,
        MINT_WITH_ADDRESS,
        SECONDARY_SALE_FEES,
    }

}

//--------------- MixedSubItem2Dto ---------------//
class MixedSubItem2Dto (
    override val parts : List<MixedPartDto>,
    val dataType : DataType?,
) : MixedOneOfItemDto() {

    enum class DataType {
        LEGACY,
        MODERN,
    }

}
