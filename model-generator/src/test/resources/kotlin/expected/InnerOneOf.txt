package com.rarible.test

import com.fasterxml.jackson.annotation.JsonSubTypes
import com.fasterxml.jackson.annotation.JsonTypeInfo
import java.math.BigInteger
import java.time.Instant
import kotlin.String

@JsonTypeInfo(use = JsonTypeInfo.Id.NAME, include = JsonTypeInfo.As.PROPERTY)
@JsonSubTypes(
    JsonSubTypes.Type(name = "sell", value = OrderActivitySellDto::class),
    JsonSubTypes.Type(name = "list", value = OrderActivityListDto::class),
    JsonSubTypes.Type(name = "bid", value = OrderActivityBidDto::class),
    JsonSubTypes.Type(name = "burn", value = NftActivityBurnDto::class),
    JsonSubTypes.Type(name = "transfer", value = NftActivityTransferDto::class),
    JsonSubTypes.Type(name = "mint", value = NftActivityMintDto::class),
)
sealed class InnerOneOfDto {
    abstract val id : String?
    abstract val date : Instant?
    abstract val address : BigInteger?
}

//--------------- OrderActivityDto ---------------//
sealed class OrderActivityDto : InnerOneOfDto() {
}

class OrderActivitySellDto (
    override val id : String?,
    override val date : Instant?,
    override val address : BigInteger?,
    val transactionHash : String?,
    val mode : Mode?,
) : OrderActivityDto()  {

    enum class Mode {
        FIRST,
        SECOND,
        THIRD,
    }

}

class OrderActivityListDto (
    override val id : String?,
    override val date : Instant?,
    override val address : BigInteger?,
    val price : BigInteger?,
    val mode : Mode?,
) : OrderActivityDto()  {

    enum class Mode {
        FIRST,
        SECOND,
        THIRD,
    }

}

class OrderActivityBidDto (
    override val id : String?,
    override val date : Instant?,
    override val address : BigInteger?,
    val price : BigInteger?,
) : OrderActivityDto()


//--------------- NftActivityDto ---------------//
sealed class NftActivityDto : InnerOneOfDto() {
    abstract val tokenId : BigInteger?
    abstract val value : BigInteger?
    abstract val transactionHash : String?
}

class NftActivityBurnDto (
    override val id : String?,
    override val date : Instant?,
    override val address : BigInteger?,
    override val tokenId : BigInteger?,
    override val value : BigInteger?,
    override val transactionHash : String?,
    val burn : String?,
    val mode : Mode?,
) : NftActivityDto()  {

    enum class Mode {
        FIRST,
        SECOND,
        THIRD,
    }

}

class NftActivityTransferDto (
    override val id : String?,
    override val date : Instant?,
    override val address : BigInteger?,
    override val tokenId : BigInteger?,
    override val value : BigInteger?,
    override val transactionHash : String?,
    val from : String?,
) : NftActivityDto()

class NftActivityMintDto (
    override val id : String?,
    override val date : Instant?,
    override val address : BigInteger?,
    override val tokenId : BigInteger?,
    override val value : BigInteger?,
    override val transactionHash : String?,
) : NftActivityDto()
