package com.rarible.test

import com.fasterxml.jackson.annotation.JsonSubTypes
import com.fasterxml.jackson.annotation.JsonTypeInfo
import java.lang.Integer
import kotlin.String

@JsonTypeInfo(use = JsonTypeInfo.Id.CUSTOM, include = JsonTypeInfo.As.PROPERTY, property = "assetClass", visible=true)
@JsonSubTypes(
    JsonSubTypes.Type(name = "SUB_ITEM_1", value = DiscriminatorSubItem1::class),
    JsonSubTypes.Type(name = "SUB_ITEM_2", value = DiscriminatorSubItem2::class),
)
sealed class DiscriminatorItem {
    abstract val assetClass : String
}

class DiscriminatorSubItem1 (
    override val assetClass : String,
    val name : String,
) : DiscriminatorItem()

class DiscriminatorSubItem2 (
    override val assetClass : String,
    val id : Integer,
) : DiscriminatorItem()
